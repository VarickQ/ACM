#include <stdio.h>
#include <string.h>
#include <math.h>

int main()
{
	int a[10],i,j,temp;
	while(scanf("%d", &a[0]) != EOF)
	{
		for( i=1 ; i < 8 ; i++ )
			scanf("%d", &a[i]);

		int sum = 0;
		for( i=0 ; i < 8 ; i++ )
		{
			for( j=0 ; j < 7 ; j++ )
				if(a[i] < a[j])
				{
					temp = a[i];
					a[i] = a[j];
					a[j] = temp;
					sum ++;
				}
		}
		/*for( i=0 ; i < 8 ; i++ )
			printf("%d ", a[i]);
		printf("\n%d\n", sum);*/
		if(sum==12)
			printf("ascending\n");
		else if(sum==28)
			printf("descending\n");
		else
			printf("mixed\n");
	}
	return 0;
}

#include <stdio.h>
#include <stdlib.h>
#define N 100001

int a[N],b[N];
int cmp ( const void *aa , const void *bb ){
    return *(int *)aa - *(int *)bb;
}

int main()
{
	int n,i,j;
	while(scanf("%d", &n) == 1)
	{
		for(i=0 ; i < n ; i++)
		{
			scanf("%d %d",&a[i],&b[i]);
			if(i==0){
				printf("%d\n",a[0]+b[0]);continue;
			}
			qsort(a,i+1,sizeof(a[0]),cmp);
			qsort(b,i+1,sizeof(b[0]),cmp);

			if(a[i] > b[i])
				printf("%d\n",a[i]+b[0]);
			else
				printf("%d\n",a[0]+b[i]);
		}
	}
	return 0;
}
