#include <stdio.h>
#include <math.h>
#include <queue>
#include <string.h>
#include <iostream>
#include <algorithm>
using namespace std;
const int inf = 100000000;
const int N = 1000005;

struct A{
	int next,v,dist;
}e[N*3];
int k,n,d[N],cnt[N];
int heade[N],qu[N*10];
bool vis[N];
void add(int u,int v,int d){
	e[k].v = v;
	e[k].dist = d;
	e[k].next = heade[u];
	heade[u] = k++;
}
int spfa(){
	for(int i=0 ; i <= n ; i++)
		d[i]=inf;
	memset(vis,0,sizeof(vis));
	d[1] = 0;
	int head=0,tail=1;
	qu[0] = 1;
	while(head < tail){
		int u = qu[head++];
		vis[u] = true;
		for(int i=heade[u] ; i!=-1 ; i=e[i].next){
			int v = e[i].v;
			if(d[v] > d[u] + e[i].dist){
				d[v] = d[u] + e[i].dist;
				if(!vis[v]){
					qu[tail++] = v;
					vis[v] = true;
					if(++cnt[v]>=n)return -1;//ÅÐ¸º»·
				}
			}
		}
		vis[u] = false;
	}
	return d[n];
}
int main()
{
	int i,ml,md,a,b,dis;
	while(scanf("%d%d%d",&n,&ml,&md)!=EOF)
	{
		memset(heade,-1,sizeof(heade));
		memset(cnt,0,sizeof(cnt));
		k = 0;
		for(i=0 ; i < ml ; i++)
		{
			scanf("%d%d%d",&a,&b,&dis);//b-a<=d
			add(a,b,dis);
		}
		for(i=0 ; i < md ; i++)
		{
			scanf("%d%d%d",&a,&b,&dis);//a-b<=-d
			add(b,a,-dis);
		}
		for(i=2 ; i <= n ; i++)
			add(i,i-1,0);
		int ans=spfa();
		if(ans==-1)printf("-1\n");
		else if(ans==inf)printf("-2\n");
		else printf("%d\n",ans);
	}
	return 0;
}
