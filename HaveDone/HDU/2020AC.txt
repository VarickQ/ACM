bool cmp(const int &a,const int &b){
	return a > b;}

int main()
{
	int va[105];
	int i,n;
	while(scanf("%d",&n) != EOF){
		if(n!=0){
		for(i=0;i<n;i++)
			scanf("%d",&va[i]);
		for(i=0;i<n;i++){
			va[i]=va[i]*10;
			if(va[i]<0){
				va[i]=-va[i];
				va[i]+=1;
			}
		}
		sort(va,va+n,cmp);
		if(va[0]%10==1)
			printf("%d",-va[0]/10);
		else
			printf("%d",va[0]/10);
		for(i=1;i<n;i++)
			if(va[i]%10==0)
				printf(" %d",va[i]/10);
			else
				printf(" %d",-va[i]/10);
		printf("\n");
		}
	}
	return 0;
}
