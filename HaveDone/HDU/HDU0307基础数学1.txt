#include <iostream>
#define max(a,b) a > b ? a : b
#define min(a,b) a < b ? a : b
using namespace std;

int main()
{
	int a,b,i,x;
	while(scanf("%d %d", &a, &b) == 2)
	{
		x = max(a,b);
		for( i=x ; ; i+=x )
			if(i%a == 0 && i%b == 0)
				break;
		printf("%d\n", i);
	}
	//system("pause");
	return 0;
}

#include <iostream>
using namespace std;

int gcd(int a,int b)
{
	if(b == 0) return a;
	else return gcd(b,a%b);
}
int main()
{
	int t,m,num,res;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &m);
		res = 1;
		while(m--)
		{
			scanf("%d", &num);
			res = num/gcd( num, res)*res;
		}
		printf("%d\n", res);
	}
	//system("pause");
	return 0;
}

using namespace std;

int dp[32768];
int main()
{
    int t,n,i,j,res;
    scanf("%d", &t);
    while(t--)
    {
        scanf("%d", &n);
        memset(dp,0,sizeof(dp));
        for( i=2,res=0 ; i <= n/2 ; i++ )
        {
            if(dp[i]==0 && n%i==0)
                for( j=i ; j < n ; j+=i )
                    dp[j] = 1;
        }
        for( i=1,res=0 ; i < n ; i++ )
            if(dp[i] == 0) res ++;
        printf("%d\n", res);
    }
    //system("pause");
    return 0;
}

#include <iostream>
#include <cmath>
using namespace std;

int fun(int n,int m){
	int i,ans=0;
	for( i=0 ; n != 0 ; i++ )
	{
		ans += n%m;
		n /= m;
	}
	return ans;
}
int main()
{
	int n;
	while(scanf("%d", &n) && n)
	{
		if( fun(n,10) == fun(n,16) && fun(n,10) == fun(n,12) && fun(n,12) == fun(n,16) )
			printf("%d is a Sky Number.\n", n);
		else
			printf("%d is not a Sky Number.\n", n);
	}
	//system("pause");
    return 0;
}
