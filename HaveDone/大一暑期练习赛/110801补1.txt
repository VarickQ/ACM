#include <string.h>
#include <math.h>
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
#define inf 100000000

int mat[105][105],h[105];
int main()
{
	int n,c,m,i,j,k,a,b,d;
	while(scanf("%d%d",&n,&c) == 2)
	{
		for(i=1 ; i<= n ; i++)
			for(j=1 ; j <= n ; j++)
				mat[i][j] = inf;
		for(i=0 ; i < c ; i++)
			scanf("%d",&h[i]);
		scanf("%d",&m);
		for(i=0 ; i < m ; i++)
		{
			scanf("%d%d%d",&a,&b,&d);
			mat[a][b] = d;
		}
		for(k=1 ; k <= n ; k++)
			for(i=1 ; i <= n ; i++)
				for(j=1 ; j <= n ; j++)
					if(mat[i][k]!=inf && mat[k][j]!=inf)
						if(mat[i][k]+mat[k][j] < mat[i][j])
							mat[i][j] = mat[i][k]+mat[k][j];
		int ans=inf,pos;
		for(i=0 ; i < c ; i++)
			for(j=1 ; j <= n ; j++)
				if(mat[h[i]][j]+mat[j][h[i]] < ans){
					ans = mat[h[i]][j]+mat[j][h[i]];
					pos = h[i];
				}
		if(ans!=inf)printf("%d\n",pos);
		else printf("I will nerver go to that city!\n");
	}
	return 0;
}

#include <stdio.h>
#include <string.h>
#include <math.h>
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
#define inf 1000000000

int mat[10][10],ans,m,n;
void dfs(int c,int temp){
	if(c==m){//这里m写成n了
		if(abs(temp) < abs(ans))
			ans = abs(temp);
	} else {
		for(int i=1 ; i < n ; i++){
			int sum1=0,sum2=0;
			for(int j=0 ; j < i ; j++)
				sum1 += mat[c][j];
			for(int j=i ; j < n ; j++)
				sum2 += mat[c][j];
			dfs(c+1,sum1-sum2+temp);
		}
	}
}
int main()
{
	int i,j,t;
	while(scanf("%d%d",&m,&n) == 2)
	{
		memset(mat,0,sizeof(mat));
		for(i=0 ; i < m ; i++)
			for(j=0 ; j < n ; j++)
				scanf("%d",&mat[i][j]);
		scanf("%d",&t);
		if(n==1){
			printf("You'd better buy another one!\n");
			continue;
		}
		ans = inf;
		int left=0,right=0;
		if(n==2){
			for(i=0 ; i < m ; i++){
				left += mat[i][0];
				right += mat[i][1];
			}
			ans = abs(left-right);
		} else {
			dfs(0,0);
		}
		if(ans <= t) printf("%d\n",ans);
		else printf("You'd better buy another one!\n");
	}
	return 0;
}
